pod.yaml

apiVersion: v1
kind: Pod
metadata:
  name: demo-pod
spec:
  containers:
    - name: nginx-container
      image: nginx
      ports:
        - containerPort: 80

minikube start


applying pod.yaml
kubectl apply -f pod.yaml


 Verifying pod status
kubectl get pods


 Describing the pod
 
kubectl describe pod demo-pod
